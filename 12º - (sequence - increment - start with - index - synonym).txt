CREATE SEQUENCE SEQ_FUNCIONARIOS
START WITH 1 --INICIA A PARTIR DO 1
INCREMENT BY 1 --INCREMENTA DE 1 EM 1
MINVALUE 1 --QUANTIDADE MINIMA A SER INCREMENTADA
MAXVALUE 1000 --QUANTIDADE MAXIMA A SER INCREMENTADA
NOCACHE --NÃO GUARDA EM CACHA A FAIXA DE VALORES
NOCYCLE --QUANTO ATINGIR OS 1000 REGISTROS, NÃO DEIXA REINICIAR O CONTADOR, APENAS PARA DE INSERIR ITENS NA TABELAS

INSERT INTO *TABELA* VALUES(SEQ_FUNCIONARIO.NEXTVAL,....)--INSERE ITULIZANDO A SEQUENCE
SELECT * FROM USER_SEQUENCES --RETORNA TODAS AS SEQUENCES CRIADAS
SELECT SEQ_FUNCIONARIO.CURRVAL FROM DUAL
ALTER SEQUENCE  SEQ_FUNCIONARIO MAXVALUE 700--REDEFINE O VALOR MAXIMO PARA 700
DROP SEQUENCE SEQ_FUNCIONARIO --EXCLUI SEQUENCE

--NÃO SE ALTERA O VALOR(ID QUE É INCREMENTADO) DA SEQUENCE APOS CRIADO
--NESTA OCASIÃO DEVE-SE EXLUIR A SEQUENCE E RECRIA-LA

CREATE INDEX IND_TABELA_COLUNA ON *TABELA*(*COLUNA*)--CRIA INDICE APENAS PARA UMA COLUNA DE UMA TABELA
SELECT *COLUNA* FROM *TABELA* WHERE *COLUNA* LIKE '%BA%'
--O INDICE SÓ FUNCIONA DE ACORDO COM A QUANTIDADE DE CASAS DO SELEC
--EX: SELECT CONTÉM 3 COLUNAS DE RETORNO? --INDICE DEVE SER CRIADO PASSANDO ESSAS MESMAS 3 TABELAS DE RETORNO, CASO CONTRÁRIO NÃO TERÁ EFEITO

CREATE INDEX IND_TABELA_COLUNAS ON *TABELA*(ITEM1, ITEM2, ITEM3...) --CRIA INDICE PARA VARIOS ITENS

SELECT * FROM USER_INDEX --RETORNA TODOS OS INDICES

DROP INDEX IND_TABELA_COLUNAS --DELETA O INDICE
