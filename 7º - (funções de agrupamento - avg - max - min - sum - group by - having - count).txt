SELECT Count(*) AS REGISTROS--CONTA A QUANTIDADE DE REGISTROS 
Avg(TOTAL) AS MEDIA, --FUNÇÃO DE MÉDIA, SE UM REGISTRO POSSUIR VALOR NULO, ELE É DISCONSIDERADO
Round(Avg(TOTAL),2) AS MEDIA, --ROUND ARREDONDA PARA 2 CASA DEPOIS DA VÍRGULA A MÉDIA FEITO PELO AVG
Max(TOTAL) AS MAXIMO, --RETORNA O MAIOR VALOR 
Min(TOTAL) AS MINIMO, --RETONA O MENOR VALOR
Sum(TOTAL) AS SOMA --SOMA TODOS OS VALORES
FROM *TABELA*

SELECT ID_ALUNO,  --UNICA CLAUSULA QUE NÃO SERÁ AGRUPADA
Count(*) AS CONTRATOS, --AGRUPA TODOS OS CONTRATOS DE CADA ID
Sum(TOTAL) AS TOTAL, --AGRUPA O VALOR TOTAL DE CADA ID
FROM *TABELA*
GROUP BY ID_ALUNO --AGRUPA MESMO ITEM DECLARADO SEM GRUPO

SELECT To_Char(Trunc(DATA), 'DD/MM/YYYY') AS DATA,--COLUNA SEM CLAUSULA DE GRUPO
Sum(TOTAL) AS SOMA, --SOMA TODOS OS VALORES
Count(*) AS QTDE --AGRUPA A QUANTIDADE DE ITENS POR DATA
FROM *TABELA*
GROUP BY Trunc(DATA) --AGRUPA PELO ITEM SEM CLUSULA DE AGRUPAMENTO

Count(DISTINCT(PESSOA)) FROM --DISTINCT NÃO DEIXA TER REGISTROS DUPLICADOS 

ESTADO, Count(*) --CONTA A QUANTIDADE DE REGISTROS POR ESTADO E AGRUPA
FROM *TABELA*
GROUP BY ESTADO --AGRUPA PELO ITEM SEM CLUSULA DE AGRUPAMENTO

HAVING --MESMA FUNÇÃO DO WHERE PORÉM É UTILIZADO PARA CLAUSULAS DE GRUPO, PODE SER DECLARADO ANTES OU DEPOIS DO GROUP BY

CODIGO, Avg(TOTAL) MEDIA
FROM *TABELA*
WHERE CODIGO > 0
HAVING Avg(TOTAL) > 500 --HAVING UTILIZADO PARA CONDIÇÕES DENTRO DE FUNÇÕES DE GRUPO
GROUP BY CODIGO
 
SELECT Max(Avg(TOTAL)) --RETORNA A MAIOR MÉDIA
FROM *TABELA*
GROUP BY ID --AGRUPADO PELO ID
