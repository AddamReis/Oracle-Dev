--CASE WHEN -- CONDICINAL
DECLARE
V_ESTADO VARCHAR(2) = 'RS';
V_NOME VARCHAR(10);
BEGIN
CASE
WHEN V_ESTADO = 'RS' THEN V_NOME := 'GAUCHO';
WHEN V_ESTADO = 'RJ' OR V_ESTADO ='ES' THEN V_NOME := 'CARIOCA';
ELSE
V_NOME := 'OUTROS';
END CASE;
Dbms_Output.Put_Line('APELIDO: '|| V_NOME);
END;

--LOOP - REPETIÇÃO
DECLARE
V_CONTADOR INTEGER := 0; --CONTADOR INICIADO COM 0
BEGIN
LOOP
V_CONTADOR := + 1; --A CADA REPETIÇÃO INCREMENTA + 1
Dbms_Output.Put_Line(V_CONTADOR); --EXIBE CADA INCREMENTAÇÃO 1,2,3...
EXIT WHEN C_CONTADOR = 10; --CONTADOR CHEGAR NO 10 É ENCERRADO
END LOOP;
Dbms_Output.Put_Line('FIM DO LOOP');
END;

--FOR LOOP
DECLARE
V_CONTADOR INTEGER;
BEGIN
FOR V_CONTADOR IN 1..10 --CONTADOR INICIA COM 1 E SE ENCERRA NO 10
LOOP --LOOP DE REPETIÇÃO
Dbms_Output.Put_Line('FOR LOOP: '||V_CONTADOR);--A CADA REPETIÇÃO SERÁ EXIBIDO UMA MENSAGEM
END LOOP;
END;

--WHILE LOOP
DECLARE
V_CONTADOR INTEGER := 0; --INICIA CONTADOR COM 0
V_TEXTO VARCHAR(10);
BEGIN
WHILE V_CONTADOR < 10 --ENQUANTO O CONTADOR FOR MENOR QUE 10 
LOOP
V_CONTADOR := V_CONTADOR +1; --INCREMENTA 1
IF(V_CONTADOR Mod 2) = 0 THEN --MOD PEGA O RESTO DA DIVISÃO
V_TEXTO:= 'PAR';
ELSE
V_TEXTO := 'IMPAR';
END IF;
Dbms_Output.Put_Line(V_CONTADOR||' --> '||V_TEXTO);
END LOOP;
END;
