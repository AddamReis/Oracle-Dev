InitCap(NOME) FROM --TRANSFORMA A PRIMEIRA LETRA DE TODAS AS PALAVRAS DA COLUNA EM MAIÚSCULO E O RESTANTE EM MINUSCULO

InStr(NOME, 'C') FROM --RETORNA A POSIÇÃO DA LETRA INFORMADA --EX: MARCIO | RETORNA: 4 --POIS A LETRA C ESTÁ NA QUARTA POSIÇÃO DO NOME

Length(NOME) FROM --INFORMA A QUANTIDADE DE CARACTERES QUE O DADO POSSUI --EX: ADDAM | RETORNA: 5 --POIS É A QUANTIDADE DE LETRAS

Lower(NOME) FROM --TRANSFORMA TODO O TEXTO EM MINÚSCULO

Upper(NOME) FROM --TRANSFORMA TODO O TEXTO EM MAIÚSCULO

LPad(SALARIO, 6, '0') FROM --PREENCHE ATÉ COMPLETAR 6 CASAS COM '0' A ESQUERDA --EX: 000025 | 001350 | 000223

RPad(SALARIO, 6, '0') FROM --PREENCHE ATÉ COMPLETAR 6 CASAS COM '0' A DIREITA --EX: 250000 | 135000 | 223000

SubStr(NOME, 1, 3) FROM --A PARTIR DA POSIÇÃO 1, COPIA OS 3 CARACTERES --EX: ADDAM | RETORNO: ADD

REPLACE(NOME, 'D', 'L') FROM --SUBSTITUI A 1º CONDIÇÃO PELA SEGUNDA --EX: ADDAM | RETORNO: ALLAM

--FUNÇÕES--
SubStr(Length(NOME), 1) FROM --RETORNA A ULTIMA LETRA DE CADA NOME, INDEPENDENTE DA QUANTIDADE DE CARACTERES --EX: ADDAM | RETORNO: M 
SubStr(Length(NOME) -1, 2) FROM --RETORNA AS DUAS ULTIMAS LETRAS DE CADA NOME, INDEPENDENTE DA QUANTIDADE DE CARACTERES --EX: ADDAM | RETORNO: AM
SubStr(3, Length(NOME) -3) FROM --RETORNA OS 3 PROXIMOS CARACTERES DO NOME A PARTIR DA TERCEIRA POSIÇÃO --EX: JULIUS | RETORNO: IUS
FROM *TABELA* WHERE Upper(SubStr(NOME, 1, 3)) = 'MAR'; --RETORNA OS DADOS CUJO AS 3 PRIMEIRAS LETRAS DO NOME SÃO 'MAR' E PASSA PRA MAIÚSCULO
