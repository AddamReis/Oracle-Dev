SELECT SYSDATE FROM DUAL --DUAL É UMA TABELA VIRTUAL DO ORACLE QUE NÃO CONTÉM NENHUMA INFORMAÇÃO

SELECT
Round(45.925, 2), --ARREDONDA PRA 2 CASAS DECIMAIS, RETORNO: 45.93
Trunc(45.929, 2), --INVERSO DE ARREDONDAMENTO, RETORNO: 45.92 
Mod(10, 2), --RETORNA O RESTO DA DIVISAO, NESTE CASO = 0
Trunc(1.99), --RETORNA NENHUMA CASA APÓS A VÍRGULA, RETORNO: 1
Trunc(1.99, 2) --RETORNA 2 CASAS DECIMAIS APÓS A VÍRGULA, RETORNO: 1,99
FROM DUAL

--DATAS--
SYSDATE + 5 FROM DUAL--RETORNA A DATA DE HOJE + 5 DIAS

Trunc(DATA - 7) FROM DUAL--RETORNA AS DATAS MENOS - 7 DIAS (NESTE CASO O TRUNC BLOQUEIA A EXIBIÇÃO DA HORA)

--HORAS--
SYSDATE + 2 / 24 FROM DUAL--DIVIDE 2 POR 24, E SOMA O RESULTADO A DATA -- ADD 2 HORAS AO SYSDATE

--MINUTOS--
SYSDATE + 15 / 1440 FROM DUAL--DIVIDE 15 POR 1440 E SOMA O RESULTADO A DATA -- ADD 15 MINUTOS
 
--SEGUNDOS--
SYSDATE + 30 / (3600 * 24) FROM DUAL--MULTIPLICA 3.600(SEGUNDOS EQUIVALENTES A 1 HORA)E MULTIPLICA POR 24 (EQUIVALE A 1 DIA) --ADD 30 SEGUNDOS

--MESES--
Months_Between(SYSDATE - 90) FROM DUAL--RETORNA A DATA MENOS 3 MESES

Add_Months(SYSDATE, 5) FROM DUAL--ADICIONA 5 MESES A DATA ATUAL

Next_Day(SYSDATE, 'SUNDAY') FROM DUAL --RETORNA A DATA DO PROXIMO SÁBADO

Last_Day(SYSDATE) FROM DUAL --RETORNA O ULTIMA DIA DO MES

Round(SYSDATE, 'MONTH') FROM DUAL -- SE O DIA FOR ATÉ 15, RETORNA O 1º DIA DO MES ATUAL, A PARTIR DO 16º RETORNA O 1º DIA DO PROXIMO MES

Trunc(SYSDATE, 'MONTH') FROM DUAL --RETORNA O PRIMEIRO DIA DO MES ATUAL

--FORMATAÇÃO DE DATA--
To_Char(SYSDATE, 'DD') FROM DUAL --RETORNA APENAS O MES CORRENTE

To_Char(SYSDATE, 'DD/MM') FROM DUAL --RETORNA MES E DIA CORRENTE

To_Char(SYSDATE, 'DD/MM/YYYY') FROM DUAL --RETORNA DIA, MES E ANO CORRENTE

To_Char(SYSDATE, 'YYYY') FROM DUAL --RETORNA ANO CORRENTE COM AS 4 CASAS

To_Char(SYSDATE, 'YY') FROM DUAL --RETORNA ANO CORRENTE COM AS 2 ULTIMAS CASAS

To_Char(SYSDATE, 'MM/YYYY') FROM DUAL --RETORNA MES E ANO CORRENTE

To_Char(SYSDATE, 'MONTH') FROM DUAL --RETORNA O MES EM TEXTT --EX: AGOSTO

To_Char(SYSDATE, 'D') FROM DUAL --RETORNA O DIA DA SEMANA -- EX: 1 = DOMINGO

To_Char(SYSDATE, 'DY') FROM DUAL --RETORNA A SIGLA DO DIA DA SEMANA --EX: TERÇA --RETORNA: TER

To_Char(SYSDATE, 'DAY') FROM DUAL --RETORNA O TEXTO DO DIA DA SEMANA --EX: QUARTA

To_Char(SYSDATE, 'YEAR') FROM DUAL --RETORNA O TEXTO DO ANO EX: TWENTY NINNETEEN (20-19)

To_Char(SYSDATE, '"SAO PAULO, " fmDAY ", " DD "de" fmMonth "de" YYYY') FROM dual--retorna a data corrente por extenso
--a sigla "fm" antes do do dia e mes, é utilizada para retirar os espaços entre os valores, assim ficando visualmente melhor

--FORMATACAO DE HORA--
To_Char(SYSDATE, 'HH24:MI') FROM DUAL --RETORNA HORA E MINUTO

To_Char(SYSDATE, 'HH24:MI:SS') FROM DUAL --RETORNA HORA, MINUTO E SEGUNDO

To_Char(SYSDATE, 'DD/MM HH24:MI') FROM DUAL --RETORNA DIA E MES, HORA E MINUTO 

To_Char(SYSDATE, 'DD/MM/YYYY HH24:MI:SS') FROM DUAL --RETORNA DATA E HORA COMPLETA
