INNER JOIN -- JUNÇÃO ENTRE DUAS TABELAS RETORNANDO O RESULTADO EXISTENTE EM AMBAS

LEFT OUTER JOIN --JUNÇÃO ENTRE DUAS TABELAS COM CLÁUSULA DE JUNÇÃO EXPLICITA, 
--PRESERVANDO LINHAS SEM CORRESPONDENCIA DA PRIMEIRA TABELA(PRESERVA TODA TABELA DA ESQUERDA)
--EX:
SELECT * FROM *TABELA1* AS 1
LEFT OUTER JOIN *TABELA2* AS 2 ON 1.ID = 2.ID
WHERE 2.DATA > '01/01/20018'
--RETORNA TODA AS LINHAS DA ESQUERDA, E TODOS DADOS INTERLIGADOS ENTRE AMBAS COM A DATA > '01/01/20018'

RIGHT OUTER JOIN  --O INVERSO DA LEFT OUTER JOIN

FULL OUTER JOIN --TRAZ OS DADOS DE AMBAS AS TABELAS SEM EXCESSÕES

CROSS JOIN --FAZ UM CARTESIANO DE DUAS TABELAS (NÃO POSSUI CLAUSULA DE JUNÇÃO
--EX 5 FUNCIONARIOS E 5 CARGOS - CADA FUNCIONARIO IRÁ POSSUIR 5 CARGOS, TOTALIZANDO 25 FUNCIONARIOS 

NATURAL JOIN -- cria um INNER ou OUTTER join entre as duas tabelas, 
--sem definição EXPLICITA do critério de junção, pois, automaticamente, se utiliza dos campos que as 
--duas tabelas tem em comum para criar a junçao (Utiliza as chaves estrangeiras que interligam as duas tabelas)
SELECT * FROM TABLE *TABELA1* NATURAL JOIN *TABLE2*

EQUIJOIN -- EQUIJOIN NADA MAIS É DO QUE O INNER JOIN IMPLICITO
--EX: 
SELECT* FROM *TABELA1* A, *TABELA2* B
WHERE A.ID = B.ID;

NO-EQUIJOIN --É TODO OPERADOR QUE FOR DIFERENTE DE IGUAL(UTILIZADO PARA INTERLIGADOR OS DADOS DE DUAS TABELAS)
--EX:
SELECT * FROM *TABELA1* AS A, *TABELA2* "B"
WHERE A.VALOR > B.VALOR
OR A.DATA_VENDA <= B.DATA_VENDA

SELF JOIN -- NADA MAIS É DO QUE A CONDIÇÃO ENTRE DUAS TABELAS
--EX:
SELECT * FROM *TABELA1* AS A, *TABELA2* AS B
WHERE A.ID = B.ID
OR A.CHASSI <> B.CHASSI
