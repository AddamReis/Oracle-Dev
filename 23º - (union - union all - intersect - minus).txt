--O COMANDO UNION É UTILIZADO PARA UNIR DOIS SELECTS OU MAIS
--UNION ELIMINA TODAS AS LINHAS DUPLICADAS
--UNION ALL RETORNA TODAS AS LINHAS, INCLUSI AS DUPLICADAS
--INTERSECT RETORDA TODAS AS LINHAS EM COMUM EM AMBOS OS CONJUNTOS
--MINUS EFETUA A SUBTRAÇÃO DE LINHAS DUPLICADAS ENTRE OS CONJUNTOS

SELECT * FROM *TABELA*
WHERE CONDIÇÃO1;

UNION --RETORNA OS DADOS DA TABELA QUE ATENDE OS SELECTS PORÉM NÃO DUPLICA A MESMA LINHA
--UNION ALL --RETORNA TODOS OS DADOS DA TABELA QUE ATENDA OS SELECTS PORÉM RETORNA LINHAS DUPLICADAS
SELECT * FROM *TABELA*
WHERE CONDIÇÃO2;

--------------------

SELECT T.*, 'UNION 1' AS QUERY FROM *TABELA* --CRIADA UMA TABELA VIRTUAL PARA SABER DE QUAL SELECT ESTÁ RETORNANDO OS DADOS
WHERE CONDIÇÃO1 

UNION ALL

SELECT T.*, 'UNION 2' AS QUERY FROM *TABELA*
WHERE CONDIÇÃO2 

UNION

SELECT T.*, 'UNION 2' AS QUERY FROM *TABELA*
WHERE CONDIÇÃO3 

--DO 1º E 2º SELECT IRÁ RETORNAR OS DADOS DUPLICADOS
ORDER BY 1 DESC
-----------------------------------------------------

SELECT * FROM *TABELA*
WHERE CONDIÇÃO1 

INTERSECT --RETORNA AMBAS AS LINHAS QUE RETORNA NOS DOIS SELECTS

SELECT * FROM *TABELA*
WHERE CONDIÇÃO2

ORDER BY 1 DESC;
