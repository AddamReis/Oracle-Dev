--SELECT

BEGIN
SELECT DATA, DATA + 10
INTO V_DT_COMPRA, V_DT_CURSO --ESSSES DOIS ITENS SÃO REFERENTES AOS 2 ITENS DO SELECT, NA MESMA ORDER EX: DATA +10 = V_DT_CURSO
FROM *TABELA*
WHERE ID_CONTRATO = &CONTRATO;
Dbms_Output.Put_Line('DATA COMPRA'||V_DT_COMPRA);
Dbms_Output.Put_Line('DATA CURSO'||V_DT_CURSO);
END;
 
--INSERT
DECLARE
VCOD *TABELA*.ID_CONTRATO%TYPE; --VARIAVEL VCOD PASSA A SER IGUAL A %TYPE
BEGIN
SELECT SEQ_CONTRATO.NEXTVAL
INTO VCOD FROM DUAL; --PEGA O PROXIMO ID A SER IMPLEMENTADO E ATRIBUI A VCOD
INSERT INTO *TABELA*(ID_CONTRATO, DATA, ID_ALUNO, DESCONTO)
VALUES (VCOD, SYSDATE, 2, NULL);
Dbms_Output.Put_Line('CRIA CONTRATO'||VCOD);
END;

--UPDATE
DECLARE
VVALOR *TABELA*.VALOR%TYPE := &VALOR;
BEGIN
UPDATE *TABELA* SET
VALOR = VALOR + VVALOR
WHERE CARGA_HORARIA >= 30
END;

--DELETE
DECLARE
VCONTRATO *TABELA*.ID_CONTRATO%TYPE := &CONTRATO;
BEGIN
DELETE FROM *TABELA*
WHERE ID_CONTRATO = VCONTRATO;
END;

DECLARE
VCONTRATO NUMBER := &ID_CONTRATO;
VTEXTO VARCHAR2(50);
BEGIN
UPDATE *TABELA* SET
DESCONTO = DESCONTO + 2
WHERE ID_CONTRATO >= VCONTRATO;
VTEXTO := SQL%ROWCOUNT; --RETORNA A QUANTIDADE DE REGISTROS AFETADOS, DEVE PERMANECER ABAIXO DO COMANDO DE ALTERAÇÃO!
Dbms_Output.Put_Line(VTEXTO||' LINHAS AFETADAS.');
END;
